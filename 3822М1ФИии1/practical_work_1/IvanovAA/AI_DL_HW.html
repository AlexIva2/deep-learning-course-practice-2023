# Задание базовой архитектуры модели
class CNN(nn.Module):
    def __init__(self):
        super(CNN, self).__init__()
        # Размер результирующей карты признаков = ((Размер входа - Размер ядра + 2 * Padding) / Stride) + 1
        # 256 x ((32 - 3 + 2*1)/1 + 1) x ((32 - 3 + 2*1)/1 + 1) = 256, 32, 32
        self.conv1 = nn.Conv2d(3, 256, kernel_size=3, stride=1, padding=1)
        self.bn1 = nn.BatchNorm2d(256)
        self.relu1 = nn.ReLU()
        # new height = (old height - kernel size) / stride + 1 = (32 -2)/2 + 1 =16 => результирующая карта признаков = 256*16*16
        self.maxpool = nn.MaxPool2d(kernel_size=2, stride=2)
        # Размер результирующей карты признаков = ((Размер входа - Размер ядра + 2 * Padding) / Stride) + 1
        # 512 x ((16 - 3 + 2*1)/1 + 1) x ((16 - 3 + 2*1)/1 + 1) = 512, 16, 16
        self.conv2 = nn.Conv2d(256, 512, kernel_size=3, stride=1, padding=1)
        self.bn2 = nn.BatchNorm2d(512)
        self.relu2 = nn.ReLU()
        # new height = (old height - kernel size) / stride + 1 = (16 -2)/2 + 1 =8 => результирующая карта признаков = 512*8*8
        self.maxpool = nn.MaxPool2d(kernel_size=2, stride=2)
        # 1024 x ((8 - 3 + 2*1)/1 + 1) x ((8 - 3 + 2*1)/1 + 1) = 1024, 8, 8
        self.conv3 = nn.Conv2d(512, 1024, kernel_size=3, stride=1, padding=1)
        self.bn3 = nn.BatchNorm2d(1024)
        self.relu3 = nn.ReLU()
        #new height = (old height - kernel size) / stride + 1 = (8-2)/2 + 1 =8 => результирующая карта признаков = 1024*4*4
        self.maxpool = nn.MaxPool2d(kernel_size=2, stride=2)
        # 2048 x ((4 - 3 + 2*1)/1 + 1) x ((4 - 3 + 2*1)/1 + 1) = 2048, 4, 4
        self.conv4 = nn.Conv2d(1024, 2048, kernel_size=3, stride=1, padding=1)
        self.bn4 = nn.BatchNorm2d(2048)
        self.relu4 = nn.ReLU()
        # new height = (old height - kernel size) / stride + 1 = (8-2)/2 + 1 =8 => результирующая карта признаков = 2048*2*2
        self.maxpool = nn.MaxPool2d(kernel_size=2, stride=2)
        self.fc = nn.Linear(2048 * 2 * 2, 10)

    def forward(self, x):
        x = self.conv1(x)
        x = self.bn1(x)
        x = self.relu1(x)
        x = self.maxpool(x)
        x = self.conv2(x)
        x = self.bn2(x)
        x = self.relu2(x)
        x = self.maxpool(x)
        x = self.conv3(x)
        x = self.bn3(x)
        x = self.relu3(x)
        x = self.maxpool(x)
        x = self.conv4(x)
        x = self.bn4(x)
        x = self.relu4(x)
        x = self.maxpool(x)
        x = x.view(x.size(0), -1)
        x = self.fc(x)
        return x
# Создание модели
model = CNN()

num_epochs = 10 # количество эпох
device = torch.device("cuda:0" if torch.cuda.is_available() else "cpu") # Выбор устройства для вычислений. Обучение построенной модели на GPU
model.to(device)

# Определение функции потерь и оптимизатора
criterion = nn.CrossEntropyLoss()
optimizer = optim.Adam(model.parameters(), lr=0.001)


train_loader = DataLoader(dataset=train_dataset, batch_size=64, shuffle=True)
test_loader = DataLoader(dataset=test_dataset, batch_size=64, shuffle=False)
epochs_time = 0

for epoch in range(num_epochs):
    start_time = time.time()
    total_loss = 0
    total_correct = 0
    total_samples = 0

    model.train()
    for images, labels in train_loader:
        images, labels = images.to(device), labels.to(device)

        optimizer.zero_grad()
        outputs = model(images)
        loss = criterion(outputs, labels)
        loss.backward()
        optimizer.step()

        total_loss += loss.item()
        _, predicted = torch.max(outputs.data, 1)
        total_samples += labels.size(0)
        total_correct += (predicted == labels).sum().item()

    epoch_train_time = time.time() - start_time
    average_loss = total_loss / len(train_loader)
    accuracy = total_correct / total_samples
    epochs_time += epoch_train_time
    print(f"Epoch {epoch+1}/{num_epochs}, Loss: {average_loss:.4f}, Accuracy: {accuracy*100:.2f}%, Epoch Time: {epoch_train_time:.2f} seconds")
print(f"Total epochs time: {epochs_time:.4f}")
